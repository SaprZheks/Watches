cmake_minimum_required(VERSION 3.5) #сообщает CMake минимальную версию, необходимую для сборки проекта. ESP-IDF предназначен для работы с CMake 3.5 или новее. Эта строка должна быть первой строкой в CMakeLists.txt файл.
include($ENV{IDF_PATH}/tools/cmake/project.cmake) #извлекает остальную функциональность CMake для настройки проекта, обнаружения всех компонентов и т. Д.
project(Wathces) #создает сам проект и указывает имя проекта. Имя проекта используется для конечных двоичных выходных файлов приложения - iemyProject.elf,myProject.bin. Для каждого файла CMakeLists можно определить только один проект.
